    // 문제 정보를 나타내는 변수
    const infoDate = "2021년 10월";
    const infoType = "정보처리기능사";
    const infoNum = "01. ";
    const infoQu = "CPU의 정보처리 속도 단위 중 초당 100만 개의 연산을 수행한다는 의미의 단위는?";

    // 선택자를 변수화해 가저다 씀
    const quizWrap = document.querySelector(".quiz__wrap");
    const quizDate = quizWrap.querySelector(".quiz__date");
    const quizType = quizWrap.querySelector(".quiz__type");
    const quizNum = quizWrap.querySelector(".quiz__question em");
    const quizQu = quizWrap.querySelector(".quiz__question span");
    const quizAnswer = quizWrap.querySelector(".quiz__answer");
    const quizDesc = quizWrap.querySelector(".quiz__desc");
    const quizConfirm = quizWrap.querySelector(".quiz__confirm");
    const quizInput = quizWrap.querySelector(".quiz__input input");
    const quizAnimation = quizWrap.querySelector(".quiz__animation");

    //문제 출력
    quizDate.innerText = infoDate;
    quizType.innerText = infoType;
    quizNum.innertText = infoNum;
    quizQu.innerText = infoQu;
    quizAnswer.innerText = infoAnswer;
    quizDesc.innerText = infoDesc;

    // 정답 해설 숨기기
    quizAnswer.classList.add("none");
    quizDesc.classList.add("none");
    
    quizConfirm.addEventListener("click", () => {
        const userAnswer = quizInput.value; // 값을 받아오는 법
        alert(userAnswer);

        if(userAnswer == infoAnswer){
            alert("정답입니다.");
            quizAnimation.classList.add("like");
        } else{
            alert("땡");
            quizAnimation.classList.add("dislike");
            quizAnswer.classList.remove("none");
            quizDesc.classList.remove("none");
        }
        
        quizConfirm.classList.add("none");
        quizInput.classList.add("none");
    });
    

    // INPUT 박스에 들어온 값을 변수로 전달받는다
    
}

                        01. 문제 정보 <br>
                         const 키워드를 통해 퀴즈 정보에 대한 값을 저장합니다. <br>

                        02. 선택자 <br>
                         querySelector를 통해 html 문서 내의 요소를 특정해 지정할 수 있습니다. <br>
                         <br><br>

                        03. 문제 출력<br>
                         querySelector를 통해 지정한 요소를 변수로 선언 하고 .innerText를 통해 특정 값을 전달합니다.<br>

                        04. 정답 해설 숨기기 <br>
                         classList.add()를 통해 지정한 구역에 class를 부여합니다. <br>

                        05. 정답 확인하기 <br>
                          element.addEventListener() 함수를 통해 특정 이벤트가 발생했을 때 지정된 함수를 실행하도록 할 수 있습니다. <br>
